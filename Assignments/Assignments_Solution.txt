class Operation {
    public static void main(String[] args) {
     int x =10;
     int y =5;
     char operation = '%';
     
     switch(operation)
    {
        case '+':
           System.out.println(x+y);
            break;

        case '-':
           System.out.println(x-y);
            break;

        case '*':
            System.out.println(x*y);
            break;

        case '/':
            System.out.println(x/y);
            break;
        case '%':
            System.out.println(x%y);
            break;

        // operator doesn't match any case constant +, -, *, /
        default:
            System.out.println("invalid operator");
    }

    }
}
